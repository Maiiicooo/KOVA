// DEZE RULES GEBRUIKEN VOOR PENDING SPOT TE APROVEN

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Spots: iedereen kan lezen en schrijven tijdelijk
    match /spots/{spotId} {
      allow read, write: if true;
    }

    // Pending spots: iedereen kan lezen en schrijven tijdelijk
    match /pending_spots/{spotId} {
      allow read, write: if true;
    }

    // Emails: laat gewoon zoals het was
    match /emails/{emailId} {
      allow read: if false;
      allow create: if true;
    }

    // Catch-all
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
